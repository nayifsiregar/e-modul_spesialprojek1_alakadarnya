Dasar Elektronika: Komponen & Rangkaian

RCL, IC Regulator, breadboard, dan pengkondisian sinyal ‚Äî fondasi yang wajib dikuasai sebelum merakit sistem nyata.

üîã
Resistor (R)
V = I √ó R (Hukum Ohm)
Menghambat arus. Melindungi LED (150‚Äì220Œ©). Membuat voltage divider. Nilai dalam Œ©, kŒ©, MŒ©. Kode warna gelang untuk identifikasi nilai.
‚ö°

üåÄ
Induktor (L)
V = L √ó (dI/dt)
Menyimpan energi dalam medan magnet. Digunakan dalam SMPS (Switching Mode Power Supply) untuk konversi tegangan efisien. Nilai dalam mH, ŒºH.
üîß
IC Regulator
LM7805: Vout = 5V tetap
LM7805: 12V ‚Üí 5V stabil. LM317: adjustable Vout. AMS1117-3.3: 5V ‚Üí 3.3V untuk sensor modern. Butuh heatsink jika dissipasi daya > 1W.
Panduan Perakitan Hardware Step-by-Step
  • 1
    Gunakan Breadboard
    Lubang horizontal tepi = jalur power (+ dan ‚àí, terhubung sepanjang baris). Lubang vertikal tengah = jalur data (terhubung per kolom 5 lubang). Tidak perlu solder untuk prototyping.
  • 2
    Konvensi Warna Kabel Jumper
    ‚ñ† Merah = VCC/5V (selalu). ‚ñ† Hitam/Gelap = GND (selalu). ‚ñ† Kuning/Orange = Sinyal data. ‚ñ† Biru = Sinyal kontrol. ‚ñ† Hijau = Sinyal sekunder. Konsistensi warna sangat membantu saat debugging.
  • 3
    Kalkulasi Resistor LED
    R = (Vsupply ‚àí Vled) / Iled. Untuk LED merah dari 5V: R = (5V ‚àí 2V) / 0.02A = 150Œ©. Gunakan 220Œ© (nilai standar terdekat yang lebih aman). Jangan pernah menghubungkan LED tanpa resistor!
  • 4
    Cek Koneksi Sebelum Power On
    Periksa tidak ada short circuit (VCC langsung ke GND). Cek orientasi LED (anoda/katoda). Cek tegangan sensor sesuai spek. Gunakan multimeter untuk verifikasi sebelum menghidupkan sistem.
  • 5
    Troubleshooting Driver Arduino
    Jika COM port tidak terbaca: buka Device Manager ‚Üí cari CH340G (atau CP2102) ‚Üí install driver dari Google. Jika upload gagal: tekan tombol Reset Arduino saat pesan "uploading" muncul di IDE. Pastikan Tools ‚Üí Board dan Port sudah benar.

/* ============================================
 * Project: Safety Interlock Iradiator
 * Standard: BAPETEN Perka No.3/2020
 * Fail-Safe Design: Default = SAFE (sumber terperisai)
 * ============================================ */
const int PIN_PINTU  = 3;   // Limit switch pintu
const int PIN_SUMBER = 2;   // LED merah = sumber radiasi
const int PIN_ALARM  = 4;   // Buzzer alarm
const int PIN_AMAN   = 5;   // LED hijau = status aman
const int PIN_SCRAM  = 6;   // Emergency stop button

bool scramActive = false;
unsigned long alarmTimer = 0;

void setup() {
  pinMode(PIN_PINTU, INPUT_PULLUP);  // LOW = tertutup (terhubung GND)
  pinMode(PIN_SCRAM, INPUT_PULLUP);
  pinMode(PIN_SUMBER, OUTPUT);
  pinMode(PIN_ALARM, OUTPUT);
  pinMode(PIN_AMAN, OUTPUT);
  Serial.begin(9600);
  
  // FAIL-SAFE: Saat startup, semua output mati (aman)
  digitalWrite(PIN_SUMBER, LOW);
  Serial.println("SISTEM INTERLOCK AKTIF");
}

void loop() {
  // Cek Emergency SCRAM (prioritas tertinggi!)
  if (digitalRead(PIN_SCRAM) == LOW) {
    scramActive = true;
    Serial.println("!! EMERGENCY SCRAM DIAKTIFKAN !!");
  }
  
  if (scramActive) {
    digitalWrite(PIN_SUMBER, LOW);
    tone(PIN_ALARM, 2000); // Alarm terus berbunyi
    digitalWrite(PIN_AMAN, LOW);
    return; // Hentikan semua proses
  }
  
  bool pintuTertutup = (digitalRead(PIN_PINTU) == LOW);
  
  if (pintuTertutup) {
    // Kondisi AMAN: sumber boleh aktif
    digitalWrite(PIN_SUMBER, HIGH);
    noTone(PIN_ALARM);
    digitalWrite(PIN_ALARM, LOW);
    digitalWrite(PIN_AMAN, HIGH);
    Serial.println("[AMAN] Iradiasi berlangsung");
  } else {
    // Kondisi BAHAYA: paksa sumber masuk (FAIL-SAFE)
    digitalWrite(PIN_SUMBER, LOW);  // ‚Üê Ini yang terpenting!
    tone(PIN_ALARM, 1500);           // Alarm berbunyi
    digitalWrite(PIN_AMAN, LOW);
    Serial.println("[BAHAYA] PINTU TERBUKA - Sumber terperisai");
  }
  delay(200);
}

const unsigned long PERIOD = 10000; // 10 detik

const float CONV_SBM20 = 0.0057; // Faktor konversi CPM‚ÜíŒºSv/h (tabung SBM-20)

void IRAM_ATTR onPulse() { counts++; }

void setup() {

  attachInterrupt(digitalPinToInterrupt(PIN_GM), onPulse, FALLING);
  pinMode(PIN_PIR, INPUT);
  pinMode(PIN_BUZZER, OUTPUT);

void loop() {
  bool adaOrang = digitalRead(PIN_PIR) == HIGH;
  if (!adaOrang) {
    lcd.setCursor(0,0); lcd.print("Menunggu Orang..");
    return;
  }
  
  unsigned long now = millis();
  long elapsed = now - startMs;
  
  // Update LCD setiap detik (non-blocking)
  static unsigned long lcdUpdate = 0;
  if (now - lcdUpdate > 1000) {
    lcdUpdate = now;
    lcd.setCursor(0,0);
    lcd.print("Cnt:"); lcd.print(counts);
    lcd.print(" T:"); lcd.print((PERIOD-elapsed)/1000); lcd.print("s  ");
  }

    float m = counts * (60000.0 / PERIOD);
    
    // Dead Time Correction: n = m / (1 - m*œÑ)
    float m_per_sec = m / 60.0; // Konversi ke CPS
    float denom = 1.0 - (m_per_sec * DEAD_TIME);

    float cpm_true = n_cps * 60.0;
    float uSv = cpm_true * CONV_SBM20;
    
    // Alarm statistik: Cacah > Background + 3‚àöBackground
    float threshold = background_cpm + 3 * sqrt(background_cpm);
    bool alarm = (cpm_true > threshold);
    
    lcd.clear();
    lcd.print("CPM:"); lcd.print(cpm_true,0);
    lcd.print(" uSv:"); lcd.print(uSv,3);

    lcd.print(alarm ? "!! KONTAMINASI !!" : "Status: BERSIH  ");
    
    digitalWrite(PIN_LED_AL, alarm);
    digitalWrite(PIN_LED_OK, !alarm);
    if (alarm) tone(PIN_BUZZER, 2000, 3000);
    
    Serial.print("CPM_true:"); Serial.print(cpm_true);
    Serial.print(",uSv/h:"); Serial.println(uSv);
    
    counts = 0;
    startMs = millis();
  }
}
